function toggleFavFilter(){ // サイクル：なし -> heart -> star -> crown -> なし if(!favFilter) favFilter = 'heart'; else if(favFilter === 'heart') favFilter = 'star'; else if(favFilter === 'star') favFilter = 'crown'; // 星の次に王冠を追加 else favFilter = null; // アイコンの色更新（王冠用のID：favFilterCrown を想定） document.getElementById('favFilterIcon').style.color = (favFilter==='heart'?'#D4537E':'#ccc'); document.getElementById('favFilterStar').style.color = (favFilter==='star'?'#BA7517':'#ccc'); // 王冠アイコンの色も変える（後ほどHTMLにIDを追加します） var crownIcon = document.getElementById('favFilterCrown'); if(crownIcon) crownIcon.style.color = (favFilter==='crown'?'#FFD700':'#ccc'); render(); }
